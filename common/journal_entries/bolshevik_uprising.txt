je_bolshevik_uprising = {
	icon = "gfx/interface/icons/event_icons/bolshevik_icon.dds"

	group = je_group_internal_affairs

	is_shown_when_inactive = {
		c:RUS = root
	}

	possible = {
		c:RUS = root
	}

	immediate = {
		set_variable = {
			name = bolshevik_revolution_var
			value = 1
		}
	}

	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					exists = c:RUS
				}
				change_variable = {
					name = bolshevik_revolution_var
					add = 1
				}
			}
		}
	}

	complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}
		NOT = {
			has_law = law_type:law_council_republic
		}
		AND = {
			custom_tooltip = {
				text = is_in_world_war
				subject = root
				has_global_variable = world_war
				global_var:world_war = {
					any_scope_play_involved = {
						this = c:RUS
					}
				}
			}
		}
	}

	on_complete = {
		trigger_event = {
			id = autocracy.1
		}
	}

	fail = {
		custom_tooltip = {
			text = has_succumbed_to_bolshevik_demands
			subject = root
			has_law = law_type:law_council_republic
		}

	}

	current_value = {
		value = root.var:bolshevik_revolution_var
	}

	goal_add_value = {
		add = 360
	}

	progressbar = yes

	weight = 100

}
